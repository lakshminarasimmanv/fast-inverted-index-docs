API Reference

Rust API​

Index​

The Index struct is the main entry point for interacting with the library.

/// Create a new index with default settings
let index = IndexBuilder::new().build()?;

/// Add a document to the index
index.add_document(1, "This is a test document")?;

/// Query the index
let query = Query::term("test");
let results = index.query(&query)?;

Index Creation​

// Create with all default options
let index = IndexBuilder::new().build()?;

// Create with custom settings
let index = IndexBuilder::new()
    .with_storage_path("/path/to/index")
    .with_cache_size(100_000)
    .with_cache_ttl(Duration::from_secs(3600))
    .with_store_positions(true)
    .with_in_memory(false)
    .build()?;

Document Operations​

// Add a document with ID and content
index.add_document(doc_id, "Document content")?;

// Add with metadata
index.add_document_with_metadata(
    doc_id,
    "Document content",
    DocumentMetadata {
        id: doc_id,
        title: "Document Title".to_string(),
        author: Some("Author Name".to_string()),
        tags: vec!["tag1".to_string(), "tag2".to_string()],
        ..Default::default()
    }
)?;

// Remove a document
index.remove_document(doc_id)?;

// Get document metadata
let docs = index.get_documents(&[doc_id])?;

Query Operations​

Basic Query API (Legacy)

// Single term query
let query = Query::term("word");
let results = index.query(&query)?;

// AND query
let query = Query::and(vec!["word1".to_string(), "word2".to_string()]);
let results = index.query(&query)?;

// OR query
let query = Query::or(vec!["word1".to_string(), "word2".to_string()]);
let results = index.query(&query)?;

// Parse query from string
let query = Query::parse("word1 word2"); // Default AND
let results = index.query(&query)?;

// Ranked search with BM25
let results = index.search_bm25(&query, None)?;

// Ranked search with TF-IDF
let results = index.search_tfidf(&query, None)?;

// Field boosting
let mut boosts = HashMap::new();
boosts.insert("title".to_string(), 2.0);
boosts.insert("tags".to_string(), 1.5);
let results = index.search_bm25(&query, Some(boosts))?;

Advanced Query Engine API

use std::ops::Bound;
use std::collections::HashMap;
use fast_inverted_index::query::{QueryNode, Value, QueryExecutionParams, ScoringAlgorithm};

// Create a term query
let query = QueryNode::term("content", "search");

// Create a phrase query
let query = QueryNode::phrase("content", vec!["fast", "search"], 2);

// Create a boosted term query
let query = QueryNode::term_with_boost("title", "search", 2.0);

// Boolean operators
let query = QueryNode::and(vec![
    QueryNode::term("content", "search"),
    QueryNode::term("title", "engine")
]);

// Range query
let query = QueryNode::range(
    "word_count", 
    Bound::Included(Value::Integer(100)), 
    Bound::Excluded(Value::Integer(500))
);

// Create execution parameters
let mut params = QueryExecutionParams {
    scoring_algorithm: ScoringAlgorithm::BM25,
    limit: Some(10),
    explain: true,
    ..Default::default()
};

// Add field boosts
params.field_boosts.insert("title".to_string(), 2.0);
params.field_boosts.insert("tags".to_string(), 1.5);

// Execute the query
let result = index.execute_query(&query, Some(params))?;

// Get scored documents
for (doc_id, score) in &result.scored_docs {
    println!("Document {}: score {}", doc_id, score);
}

// Get score explanations
if let Some(explanations) = &result.explanations {
    if let Some(explanation) = explanations.get(&doc_id) {
        println!("{}", explanation.to_string_tree());
    }
}

See the Query Engine documentation for more details on the advanced query capabilities.

Term Suggestions​

// Get term suggestions for a prefix
let suggestions = index.suggest_terms("prog")?;

// Access suggestion details
for suggestion in suggestions {
    println!("Term: {}, Score: {}, Doc Frequency: {}", 
        suggestion.term, suggestion.score, suggestion.doc_frequency);
}

Index Management​

// Get index statistics
let stats = index.stats()?;
println!("Documents: {}, Terms: {}", stats.document_count, stats.term_count);

// Optimize the index
index.optimize()?;

// Create a backup
index.backup(PathBuf::from("/path/to/backup"))?;

Error Handling​

use fast_inverted_index::errors::{Result, IndexError};

fn example() -> Result<()> {
    let index = IndexBuilder::new().build()?;
    
    // Handle specific error types
    match index.query(&Query::term("test")) {
        Ok(results) => println!("Found {} results", results.doc_ids.len()),
        Err(IndexError::TermNotFound(term)) => println!("Term not found: {}", term),
        Err(IndexError::IoError(e)) => println!("IO error: {}", e),
        Err(e) => println!("Other error: {}", e),
    }
    
    Ok(())
}

Python API​

Index Creation​

Basic Construction (Simple)​

from fast_inverted_index import Index, Schema

# Create with default settings
index = Index()

# Create with custom settings
index = Index(
    storage_path="/path/to/index",
    cache_size=100000,
    cache_ttl_secs=3600,
    store_positions=True,
    in_memory=False
)

# Create with a schema
schema = Schema()
schema.add_field(FieldSchema.text("title").with_boost(5.0))
schema.add_field(FieldSchema.text("content").with_boost(1.0))
schema.set_default_field("content")

index_with_schema = Index(
    storage_path="/path/to/index",
    schema=schema
)

Builder Pattern (Advanced)​

For more complex configuration needs, use the builder pattern:

from fast_inverted_index import Index, IndexBuilder, DictionaryConfig, Schema

# Create using builder pattern
builder = Index.builder()
builder.with_in_memory(True)
builder.with_cache_size(50000)
builder.with_cache_ttl_secs(3600)
builder.with_store_positions(True)
index = builder.build()

# With dictionary configuration
dict_config = DictionaryConfig(
    hot_terms_capacity=1000,
    hot_promotion_threshold=50,
    medium_promotion_threshold=5,
    optimization_interval_secs=1800
)

# With schema
schema = Schema()
schema.add_field(FieldSchema.text("title").with_boost(5.0))
schema.add_field(FieldSchema.text("content").with_boost(1.0))

advanced_builder = Index.builder()
advanced_builder.with_dictionary_config(dict_config)
advanced_builder.with_schema(schema)
advanced_index = advanced_builder.build()

Using IndexConfig (Advanced)​

Another approach for advanced configuration:

from fast_inverted_index import Index, IndexConfig, DictionaryConfig, Schema

# Create dictionary config
dict_config = DictionaryConfig(
    hot_terms_capacity=1000,
    hot_promotion_threshold=50
)

# Create schema
schema = Schema()
schema.add_field(FieldSchema.text("title").with_boost(5.0))
schema.add_field(FieldSchema.text("content").with_boost(1.0))

# Create index config
config = IndexConfig(
    in_memory=True,
    cache_size=100000,
    store_positions=True,
    dictionary_config=dict_config,
    schema=schema
)

# Create index with config
index = Index.with_config(config)

Basic Document Operations​

# Add a simple document (legacy method)
index.add_document(
    doc_id=1,
    content="This is a test document"
)

# Add with metadata (legacy method)
index.add_document(
    doc_id=2,
    content="Another document with metadata",
    metadata={
        "title": "Document Title",
        "author": "Author Name",
        "tags": ["tag1", "tag2"],
        "category": "Example",
        "language": "en",
        "created_at": "2023-01-01T00:00:00",
        "source_url": "https://example.com/doc"
    }
)

# Remove a document
index.remove_document(1)

# Get a document
doc = index.get_document(2)
print(f"Title: {doc['title']}")

# Get multiple documents
docs = index.get_documents([1, 2, 3])

Multi-Field Document Operations​

The preferred method for working with multi-field documents is add_document_with_fields:

# Create an index with schema
schema = Schema()
schema.add_field(FieldSchema.text("title").with_boost(5.0))
schema.add_field(FieldSchema.text("content").with_boost(1.0))
schema.add_field(FieldSchema.keyword("tags").with_boost(3.0))
schema.add_field(FieldSchema.text("author").with_boost(2.0))
schema.set_default_field("content")

index = Index(storage_path="/path/to/index", schema=schema)

# Add a document with fields
doc_fields = {
    "title": "Python Programming Guide",
    "content": "Python is a high-level programming language with simple, easy-to-learn syntax.",
    "tags": "programming python tutorial",  # For keyword fields, strings are used for terms
    "author": "Jane Smith"
}
index.add_document_with_fields(1, doc_fields)

# Get a document with fields
doc = index.get_document(1)
print(f"Title: {doc['title']}")
print(f"Author: {doc['author']}")
print(f"Tags: {doc['tags']}")  # Will be returned as a list of strings

Query Operations​

Basic Query API

# Single term query
results = index.term_query("test")

# AND query
results = index.and_query(["word1", "word2"])

# OR query
results = index.or_query(["word1", "word2"])

# Parse and execute a query string
results = index.parse_query("word1 word2")  # Default AND

# Get term frequencies
freqs = index.get_term_frequencies(["word1", "word2"])

# Advanced search with ranking
results = index.search("test document", ranking_method="bm25")

# Search with field boosting and full document data
results = index.search_with_metadata(
    "test document",
    ranking_method="bm25",
    boost_fields={"title": 2.0, "tags": 1.5},
    limit=10
)

Advanced Query Engine API

from fast_inverted_index import PyQueryNode, PyValue, PyQueryBound, PyQueryExecutionParams

# Create a term query
query = PyQueryNode.term("content", "search")

# Create a boosted query
query = PyQueryNode.term("title", "search", boost=2.0)

# Create a phrase query
query = PyQueryNode.phrase("content", ["fast", "search"], 2)

# Boolean operators
query = PyQueryNode.and([
    PyQueryNode.term("content", "search"),
    PyQueryNode.term("title", "engine")
])

# Range query for numeric field
start_bound = PyQueryBound.included(PyValue.integer(100))
end_bound = PyQueryBound.excluded(PyValue.integer(500))
query = PyQueryNode.range("word_count", start_bound, end_bound)

# Create execution parameters
params = PyQueryExecutionParams(
    scoring_algorithm="bm25l",
    explain=True,
    limit=10,
    bm25_k1=1.2,
    bm25_b=0.75,
    field_boosts={"title": 2.0, "tags": 1.5}
)

# Execute the query
result = index.execute_query(query, params)

# Get results
for doc_id, score in result.scored_docs:
    doc = index.get_document(doc_id)
    print(f"Document: {doc['title']}, Score: {score}")

# Get explanations
if result.explanations:
    for doc_id, explanation in result.explanations.items():
        print(f"Explanation for document {doc_id}:")
        print(explanation.to_tree())

Field-Specific Search Helpers

# Search in a specific field
results = index.search_field("title", "search", algorithm="bm25", limit=10)

# Search for a phrase
results = index.search_phrase("content", ["fast", "search"], slop=2, 
                             algorithm="bm25", limit=10)

# Search multiple fields with different boosts
field_values = [
    ("title", "search", 2.0),
    ("content", "engine", 1.0),
    ("tags", "search", 1.5)
]
results = index.search_multi_field(field_values, algorithm="bm25", limit=10)

    doc = index.get_document(999)

except IndexError as e:
    print(f"Index error: {e}")
