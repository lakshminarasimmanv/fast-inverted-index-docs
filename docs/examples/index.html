Usage Examples

This section provides practical examples of using Fast-Inverted-Index for real-world scenarios.

Basic Search Engineâ€‹

Create a complete search engine with indexing and querying capabilities:

from fast_inverted_index import Index
import os

class SimpleSearchEngine:
    def __init__(self, storage_path=None, in_memory=False):
        """Initialize the search engine."""
        self.index = Index(
            storage_path=storage_path,
            in_memory=in_memory,
            cache_size=10000,
        )
        self.next_doc_id = 1

    def index_file(self, file_path):
        """Index a text file."""
        doc_id = self.next_doc_id
        self.next_doc_id += 1

        with open(file_path, "r", encoding="utf-8") as f:
            content = f.read()

        # Extract metadata from the file
        metadata = {
            "title": os.path.basename(file_path),
            "source_url": file_path,
            "language": "en",
        }

        # Add the document to the index
        self.index.add_document(
            doc_id=doc_id,
            content=content,
            metadata=metadata,
        )

        return doc_id

    def index_directory(self, directory_path, file_pattern=r".*\.txt$"):
        """Index all text files in a directory."""
        import re
        doc_ids = []
        file_regex = re.compile(file_pattern)

        for root, _, files in os.walk(directory_path):
            for file in files:
                if file_regex.match(file):
                    file_path = os.path.join(root, file)
                    doc_id = self.index_file(file_path)
                    doc_ids.append(doc_id)

        return doc_ids

    def search(self, query):
        """Search the index."""
        # Split the query into words
        words = query.lower().split()
        
        try:
            # For single term queries
            if len(words) == 1:
                doc_ids = self.index.term_query(words[0])
            # For multi-term queries - use AND by default
            else:
                doc_ids = self.index.and_query(words)
            
            # Get documents for the matching IDs
            results = []
            for doc_id in doc_ids:
                try:
                    doc = self.index.get_document(doc_id)
                    results.append(doc)
                except Exception:
                    pass
                    
            return results
        except Exception as e:
            print(f"Search failed: {e}")
            return []

    def suggest(self, prefix):
        """Get term suggestions."""
        return self.index.suggest_terms(prefix)

    def stats(self):
        """Get index statistics."""
        return self.index.stats()

    def optimize(self):
        """Optimize the index."""
        self.index.optimize()


# Usage example
if __name__ == "__main__":
    import sys
    
    # Create the search engine
    engine = SimpleSearchEngine(in_memory=True)
    
    # Check if a directory was provided
    if len(sys.argv) > 1:
        directory = sys.argv[1]
        print(f"Indexing directory: {directory}")
        doc_ids = engine.index_directory(directory)
        print(f"Indexed {len(doc_ids)} documents")
    
    # Interactive search loop
    print("Enter a query (or 'quit' to exit):")
    while True:
        query = input("> ")
        if query.lower() == "quit":
            break
        elif query.lower() == "stats":
            stats = engine.stats()
            for key, value in stats.items():
                print(f"{key}: {value}")
        elif query.lower().startswith("suggest "):
            prefix = query[8:].strip()
            suggestions = engine.suggest(prefix)
            for suggestion in suggestions:
                print(f"{suggestion['term']} (score: {suggestion['score']})")
        else:
            results = engine.search(query)
            print(f"Found {len(results)} results:")
            for i, doc in enumerate(results[:10], 1):
                print(f"{i}. {doc['title']} (ID: {doc['id']})")

from fast_inverted_index import Index
import requests
from bs4 import BeautifulSoup
import json
import time

class WikipediaIndexer:
    def __init__(self, storage_path=None, in_memory=False):
        """Initialize the indexer."""
        self.index = Index(
            storage_path=storage_path,
            in_memory=in_memory,
            cache_size=100000,
            cache_ttl_secs=3600,
        )
        self.next_doc_id = 1

    def index_article(self, article):
        """Index a Wikipedia article."""
        doc_id = self.next_doc_id
        self.next_doc_id += 1

        # Extract article metadata
        title = article.get("title", "")
        text = article.get("text", "")
        url = article.get("url", "")
        categories = article.get("categories", [])

        # Clean the text
        if isinstance(text, str):
            # Remove HTML tags
            soup = BeautifulSoup(text, "html.parser")
            text = soup.get_text()
        else:
            text = ""

        # Add the document to the index
        self.index.add_document(
            doc_id=doc_id,
            content=text,
            metadata={
                "title": title,
                "source_url": url,
                "category": ", ".join(categories) if categories else None,
                "language": "en",
                "tags": categories if categories else [],
            },
        )

        return doc_id

    def search(self, query, limit=10):
        """Search the index."""
        # Split the query into words
        words = query.lower().split()
        
        try:
            # For single term queries
            if len(words) == 1:
                doc_ids = self.index.term_query(words[0])
            # For multi-term queries - use AND by default
            else:
                doc_ids = self.index.and_query(words)
            
            # Limit the number of results
            if limit > 0:
                doc_ids = doc_ids[:limit]
            
            # Get documents for the matching IDs
            results = []
            for doc_id in doc_ids:
                try:
                    doc = self.index.get_document(doc_id)
                    results.append(doc)
                except Exception:
                    pass
                    
            return results
        except Exception as e:
            print(f"Search failed: {e}")
            return []

    def download_sample_articles(self, count=100):
        """Download sample Wikipedia articles."""
        articles = []
        batch_size = min(count, 20)  # Wikipedia API limits
        continue_params = {}

        while len(articles) < count:
            # Prepare the API request
            params = {
                "action": "query",
                "format": "json",
                "list": "random",
                "rnnamespace": 0,  # Main namespace
                "rnlimit": batch_size,
            }
            params.update(continue_params)

            # Make the API request
            response = requests.get(
                "https://en.wikipedia.org/w/api.php",
                params=params,
                headers={"User-Agent": "WikipediaIndexerExample/1.0"},
            )
            data = response.json()

            # Process the results
            for page in data.get("query", {}).get("random", []):
                page_id = page.get("id")
                title = page.get("title")

                # Get the article content
                content_params = {
                    "action": "query",
                    "format": "json",
                    "prop": "extracts|categories|info",
                    "exintro": 1,
                    "explaintext": 1,
                    "inprop": "url",
                    "pageids": page_id,
                }

                content_response = requests.get(
                    "https://en.wikipedia.org/w/api.php",
                    params=content_params,
                    headers={"User-Agent": "WikipediaIndexerExample/1.0"},
                )
                content_data = content_response.json()

                # Extract the article data
                page_data = content_data.get("query", {}).get("pages", {}).get(str(page_id), {})
                extract = page_data.get("extract", "")
                url = page_data.get("fullurl", "")
                categories = [
                    cat.get("title", "").replace("Category:", "")
                    for cat in page_data.get("categories", [])
                ]

                # Add the article to the list
                articles.append({
                    "id": page_id,
                    "title": title,
                    "text": extract,
                    "url": url,
                    "categories": categories,
                })

                if len(articles) >= count:
                    break

            # Check if there are more results
            if "continue" not in data:
                break
            continue_params = data.get("continue", {})

        return articles

Real-time Search with Indexing Updatesâ€‹

This example demonstrates how to build a search engine that updates its index in real-time:

import threading
import time
import random
from fast_inverted_index import Index

class RealtimeSearchEngine:
    def __init__(self, storage_path=None):
        """Initialize the real-time search engine."""
        self.index = Index(storage_path=storage_path)
        self.lock = threading.Lock()
        self.next_doc_id = 1
        self.stop_requested = False
        self.indexer_thread = None
        
    def start_indexer(self, interval=1.0):
        """Start the background indexer thread."""
        if self.indexer_thread is not None:
            return
            
        self.stop_requested = False
        self.indexer_thread = threading.Thread(
            target=self._indexer_loop,
            args=(interval,)
        )
        self.indexer_thread.daemon = True
        self.indexer_thread.start()
        
    def stop_indexer(self):
        """Stop the background indexer thread."""
        self.stop_requested = True
        if self.indexer_thread:
            self.indexer_thread.join()
            self.indexer_thread = None
            
    def _indexer_loop(self, interval):
        """Background thread that simulates incoming documents."""
        while not self.stop_requested:
            # Simulate a new document arriving
            self.add_document(self._generate_random_document())
            time.sleep(interval)
            
    def _generate_random_document(self):
        """Generate a random document for testing."""
        topics = ["Python", "Rust", "Java", "JavaScript", "Go", "C++", "Ruby"]
        actions = ["Programming", "Development", "Coding", "Learning", "Teaching"]
        qualities = ["Fast", "Efficient", "Modern", "Reliable", "Scalable"]
        
        topic = random.choice(topics)
        action = random.choice(actions)
        quality = random.choice(qualities)
        
        return {
            "content": f"{topic} is great for {action}. It's a {quality} language.",
            "metadata": {
                "title": f"{topic} for {action}",
                "tags": [topic.lower(), action.lower(), quality.lower()],
                "category": "Programming Languages",
                "timestamp": time.time()
            }
        }
        
    def add_document(self, doc_data):
        """Add a document to the index."""
        with self.lock:
            doc_id = self.next_doc_id
            self.next_doc_id += 1
            
            self.index.add_document(
                doc_id=doc_id,
                content=doc_data["content"],
                metadata=doc_data["metadata"]
            )
            return doc_id
            
    def search(self, query):
        """Search the index with the given query."""
        with self.lock:
            # Use the search_with_metadata method for rich results
            return self.index.search_with_metadata(
                query, 
                ranking_method="bm25",
                limit=10
            )
            
    def stats(self):
        """Get current statistics."""
        with self.lock:
            return self.index.stats()


# Usage example
if __name__ == "__main__":
    # Create the search engine
    search_engine = RealtimeSearchEngine()
    
    # Start the background indexer
    search_engine.start_indexer(interval=0.5)  # Add a document every 0.5 seconds
    
    try:
        # Main loop for searching
        print("Real-time search engine started. Enter queries or 'quit' to exit.")
        print("Documents are being added in the background...")
        
        while True:
            query = input("> ")
            if query.lower() == 'quit':
                break
            elif query.lower() == 'stats':
                stats = search_engine.stats()
                print(f"Documents: {stats['document_count']}")
                print(f"Terms: {stats['term_count']}")
                print(f"Cache hit rate: {stats['cache_hit_rate']}")
            else:
                start_time = time.time()
                results = search_engine.search(query)
                elapsed = time.time() - start_time
                
                print(f"Found {len(results)} results in {elapsed:.4f} seconds:")
                for i, doc in enumerate(results, 1):
                    print(f"{i}. {doc['title']} (Score: {doc['score']:.4f})")
                    print(f"   Tags: {', '.join(doc['tags'])}")
                    
    finally:
        # Clean up
        search_engine.stop_indexer()
        print("Search engine stopped.")

Benchmarking Performanceâ€‹

This example demonstrates how to benchmark indexing and query performance:

from fast_inverted_index import Index
import time
import random
import string
import multiprocessing
import statistics

def generate_random_text(word_count, max_word_length=10):
    """Generate random text with the specified number of words."""
    words = []
    for _ in range(word_count):
        word_length = random.randint(3, max_word_length)
        word = ''.join(random.choice(string.ascii_lowercase) for _ in range(word_length))
        words.append(word)
    return ' '.join(words)

def generate_random_documents(count, min_words=50, max_words=200):
    """Generate a list of random documents."""
    documents = []
    for i in range(count):
        word_count = random.randint(min_words, max_words)
        documents.append({
            "id": i + 1,
            "content": generate_random_text(word_count)
        })
    return documents

def benchmark_indexing(index, documents):
    """Benchmark document indexing performance."""
    start_time = time.time()
    for doc in documents:
        index.add_document(doc["id"], doc["content"])
    end_time = time.time()
    
    total_time = end_time - start_time
    docs_per_second = len(documents) / total_time
    
    return {
        "total_time": total_time,
        "docs_per_second": docs_per_second,
        "avg_time_per_doc": total_time / len(documents)
    }

def benchmark_querying(index, queries, repetitions=5):
    """Benchmark query performance."""
    results = []
    
    for query in queries:
        query_times = []
        
        for _ in range(repetitions):
            start_time = time.time()
            index.term_query(query)
            end_time = time.time()
            query_times.append(end_time - start_time)
        
        avg_time = statistics.mean(query_times)
        results.append({
            "query": query,
            "avg_time": avg_time,
            "min_time": min(query_times),
            "max_time": max(query_times),
            "stddev": statistics.stdev(query_times) if len(query_times) > 1 else 0
        })
    
    return results

def extract_queries_from_documents(documents, count=100):
    """Extract random words from documents to use as queries."""
    all_words = set()
    for doc in documents:
        words = doc["content"].split()
        all_words.update(words)
    
    return random.sample(list(all_words), min(count, len(all_words)))

def parallel_indexing_benchmark(processes, docs_per_process, min_words, max_words):
    """Benchmark indexing with multiple processes."""
    results = []
    
    def worker(docs_count, min_w, max_w):
        index = Index(in_memory=True)
        docs = generate_random_documents(docs_count, min_w, max_w)
        result = benchmark_indexing(index, docs)
        return result
    
    with multiprocessing.Pool(processes) as pool:
        args = [(docs_per_process, min_words, max_words) for _ in range(processes)]
        results = pool.starmap(worker, args)
    
    # Aggregate results
    total_docs = processes * docs_per_process
    total_time = sum(r["total_time"] for r in results)
    avg_time_per_doc = statistics.mean(r["avg_time_per_doc"] for r in results)
    total_docs_per_second = total_docs / total_time
    
    return {
        "processes": processes,
        "total_docs": total_docs,
        "total_time": total_time,
        "avg_time_per_doc": avg_time_per_doc,
        "total_docs_per_second": total_docs_per_second
    }

# Run the benchmarks
if __name__ == "__main__":
    # Parameters
    doc_count = 10000
    query_count = 100
    
    # Create the index
    print(f"Creating index and generating {doc_count} random documents...")
    index = Index(in_memory=True)
    documents = generate_random_documents(doc_count)
    
    # Benchmark indexing
    print("Benchmarking indexing performance...")
    indexing_results = benchmark_indexing(index, documents)
    print(f"Indexed {doc_count} documents in {indexing_results['total_time']:.2f} seconds")
    print(f"Indexing speed: {indexing_results['docs_per_second']:.2f} docs/second")
    
    # Get index statistics
    stats = index.stats()
    print("\nIndex statistics:")
    print(f"Documents: {stats['document_count']}")
    print(f"Terms: {stats['term_count']}")
    print(f"Average document length: {stats['avg_doc_length']:.2f} tokens")
    
    # Benchmark querying
    print("\nBenchmarking query performance...")
    queries = extract_queries_from_documents(documents, query_count)
    query_results = benchmark_querying(index, queries)
    
    avg_query_time = statistics.mean(r["avg_time"] for r in query_results)
    print(f"Average query time: {avg_query_time * 1000:.2f} ms")
    print(f"Queries per second: {1.0 / avg_query_time:.2f}")
    
    # Benchmark parallel indexing
    print("\nBenchmarking parallel indexing...")
    cpu_count = multiprocessing.cpu_count()
    docs_per_process = 1000
    
    for num_processes in [1, 2, 4, max(1, cpu_count-1)]:
        print(f"\nTesting with {num_processes} processes...")
        parallel_results = parallel_indexing_benchmark(
            num_processes, docs_per_process, 50, 200)
        print(f"Total docs: {parallel_results['total_docs']}")
        print(f"Total time: {parallel_results['total_time']:.2f} seconds")
        print(f"Throughput: {parallel_results['total_docs_per_second']:.2f} docs/second")

    else:
        content = f"This advanced technical article explores complex programming patterns and algorithms in {word_count} words."
    
    # Add the document
    index.add_document(
        doc_id=i,
        content=content,
        metadata={
            "title": f"Technical Article {i}",
            "level": level,
            "metadata": {
                "reading_time": reading_time,
                "word_count": word_count
            }
        }
    )

# Create a query for beginner-friendly articles with 10-20 minute reading time
# 1. Level filter
level_query = PyQueryNode.term("level", "Beginner")

# 2. Reading time range (10-20 minutes)
min_time = PyQueryBound.included(PyValue.integer(10))
max_time = PyQueryBound.excluded(PyValue.integer(21))
# Note: Use direct field name "reading_time", not "metadata.reading_time"
time_query = PyQueryNode.range("reading_time", min_time, max_time)

# 3. Content relevance
content_query = PyQueryNode.term("content", "programming")

# Combine: content query with filters for level and reading time
final_query = PyQueryNode.filter(
    content_query,
    PyQueryNode.and([level_query, time_query])
)

# Execute the query
result = index.execute_query(final_query)

# Display results
print("Beginner-friendly programming articles with 10-20 minute reading time:\n")
for doc_id, score in result.scored_docs:
    doc = index.get_document(doc_id)
    print(f"Article: {doc['title']}")
    print(f"Level: {doc['level']}")
    print(f"Reading time: {doc['metadata']['reading_time']} minutes")
    print(f"Word count: {doc['metadata']['word_count']} words")
    print(f"Score: {score:.4f}")
    print()
