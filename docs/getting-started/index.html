Getting Started with Fast-Inverted-Index

This guide will help you get started with the Fast-Inverted-Index library for building high-performance search functionality in your applications.

Installation​

Python​

Install from PyPI:

pip install fast-inverted-index

Rust​

Add to your Cargo.toml:

[dependencies]
fast-inverted-index = "0.4.6"

Prerequisites​

  • Rust 1.58 or later (if using Rust API)
  • Python 3.7 or later (if using Python API)
  • For Python, the library includes pre-built wheels for common platforms
  • For building from source, you'll need a C++ compiler and RocksDB development libraries

Quickstart​

Basic Python Example​

Create a simple search engine in Python:

from fast_inverted_index import Index

# Create an index
index = Index(in_memory=True)

# Add some documents
index.add_document(1, "Python is a programming language")
index.add_document(2, "Rust is a systems programming language")
index.add_document(3, "Search engines use inverted indices")

# Search for documents
results = index.search("programming")
print(f"Found {len(results)} results for 'programming'")

# Print the matching document IDs and scores
for doc_id, score in results:
    print(f"Document ID: {doc_id}, Score: {score:.4f}")
    doc = index.get_document(doc_id)
    print(f"Metadata: {doc}")

# Get term suggestions
suggestions = index.suggest_terms("pro")
print("Suggestions for 'pro':")
for suggestion in suggestions:
    print(f"  {suggestion['term']} (score: {suggestion['score']})")

Python Example with Redis Content Storage​

Create a search engine with Redis for content storage:

import json
import redis
from fast_inverted_index import Index

# Connect to Redis
redis_client = redis.Redis(host='localhost', port=6379, db=0)
redis_prefix = "search:"

# Create an index
index = Index(in_memory=True)

# Function to add a document with Redis content storage
def add_document(doc_id, content, metadata=None):
    # Add to index for searching
    index.add_document(doc_id, content, metadata or {})
    
    # Store full content in Redis
    redis_client.set(
        f"{redis_prefix}doc:{doc_id}",
        json.dumps({"content": content, "metadata": metadata or {}})
    )

# Function to retrieve a document with content
def get_document(doc_id):
    # Get metadata from index
    doc_metadata = index.get_document(doc_id)
    
    # Get content from Redis
    redis_data = redis_client.get(f"{redis_prefix}doc:{doc_id}")
    if redis_data:
        content_data = json.loads(redis_data)
        # Combine metadata and content
        return {**doc_metadata, "content": content_data["content"]}
    
    return doc_metadata  # Fall back to just metadata

# Add some documents
add_document(
    1, 
    "Python is a programming language with simple, easy-to-learn syntax.",
    {"title": "Introduction to Python", "tags": ["python", "programming"]}
)

add_document(
    2, 
    "Rust provides memory safety without garbage collection, making it ideal for systems programming.",
    {"title": "Rust Systems Programming", "tags": ["rust", "programming", "systems"]}
)

# Search for documents
results = index.search("programming")
print(f"Found {len(results)} results for 'programming'")

# Retrieve and display results with content
for doc_id, score in results:
    # Get document with content from Redis
    doc = get_document(doc_id)
    
    print(f"\nDocument ID: {doc_id}, Score: {score:.4f}")
    print(f"Title: {doc.get('title', 'No title')}")
    print(f"Tags: {', '.join(doc.get('tags', []))}")
    
    # Display content snippet
    content = doc.get("content", "")
    if len(content) > 100:
        content = content[:100] + "..."
    print(f"Content: {content}")

Multi-Field Python Example​

Create a search engine with field-specific indexing and searching:

from fast_inverted_index import Index, Schema, FieldSchema, QueryNode, QueryExecutionParams

# Create a schema with different field types
schema = Schema()
schema.add_field(FieldSchema.text("title").with_boost(5.0))    # Title field gets 5x boost
schema.add_field(FieldSchema.text("content").with_boost(1.0))  # Content field has normal importance
schema.add_field(FieldSchema.keyword("tags").with_boost(3.0))  # Tags field gets 3x boost, uses keyword analyzer
schema.set_default_field("content")  # Default field for queries that don't specify a field

# Create an index with the schema
index = Index(in_memory=True, schema=schema)

# Add documents with fields
doc1_fields = {
    "title": "Introduction to Python",
    "content": "Python is a high-level programming language with simple syntax.",
    "tags": "programming python tutorial"
}
index.add_document_with_fields(1, doc1_fields)

doc2_fields = {
    "title": "Rust Systems Programming",
    "content": "Rust provides memory safety without garbage collection.",
    "tags": "programming rust systems"
}
index.add_document_with_fields(2, doc2_fields)

doc3_fields = {
    "title": "Search Engine Technology",
    "content": "Search engines commonly use inverted indices for efficient retrieval.",
    "tags": "search indexing information-retrieval"
}
index.add_document_with_fields(3, doc3_fields)

# Search in a specific field
title_results = index.search_field("title", "Python", ranking_method="bm25", limit=10)
print(f"Documents with 'Python' in title: {len(title_results)}")

# Search across all fields with different boosts
boost_fields = {
    "title": 5.0,    # Title matches are 5x more important
    "content": 1.0,  # Content matches have normal importance
    "tags": 3.0      # Tag matches are 3x more important
}
results = index.search("programming", ranking_method="bm25", boost_fields=boost_fields)
print(f"Documents about 'programming' across all fields: {len(results)}")

# Advanced query using QueryNode API
query = QueryNode.OR([
    QueryNode.term("title", "search", boost=5.0),    # Search in title with 5x boost
    QueryNode.term("content", "inverted", boost=1.0),  # Search in content with normal importance
    QueryNode.term("tags", "indexing", boost=3.0)    # Search in tags with 3x boost
])

params = QueryExecutionParams(
    scoring_algorithm="bm25",
    limit=10
)

query_results = index.execute_query(query, params)
print(f"Documents matching complex query: {len(query_results.scored_docs)}")

# Get the first matching document
for doc_id, score in query_results.scored_docs:
    doc = index.get_document(doc_id)
    print(f"Document ID: {doc_id}, Score: {score:.4f}")
    print(f"Title: {doc.get('title', 'Untitled')}")
    print(f"Tags: {doc.get('tags', [])}")
    break

Basic Rust Example​

Create a search engine in Rust:

use fast_inverted_index::{IndexBuilder, Query};
use std::error::Error;

fn main() -> Result<(), Box<dyn Error>> {
    // Create an in-memory index
    let mut index = IndexBuilder::new()
        .with_in_memory(true)
        .build()?;
    
    // Add some documents
    index.add_document(1, "Rust is a systems programming language")?;
    index.add_document(2, "Fast and reliable code with memory safety")?;
    index.add_document(3, "Search engines commonly use inverted indices")?;
    
    // Search for documents
    let query = Query::term("programming");
    let results = index.query(&query)?;
    
    println!("Found {} documents containing 'programming'", results.doc_ids.len());
    println!("Document IDs: {:?}", results.doc_ids);
    
    // Get term suggestions
    let suggestions = index.suggest_terms("pro")?;
    println!("Suggestions for 'pro':");
    for suggestion in suggestions {
        println!("  {} (score: {})", suggestion.term, suggestion.score);
    }
    
    Ok(())
}

Multi-Field Rust Example​

Create a search engine with field-specific indexing and searching in Rust:

use fast_inverted_index::{
    IndexBuilder,
    schema::{Schema, SchemaBuilder, FieldSchema, FieldType, AnalyzerType},
    query::{QueryNode, QueryExecutionParams, ScoringAlgorithm}
};
use std::collections::HashMap;
use std::error::Error;

fn main() -> Result<(), Box<dyn Error>> {
    // Create a schema with different field types
    let schema = SchemaBuilder::new()
        .add_field(FieldSchema::text("title").with_boost(5.0))
        .add_field(FieldSchema::text("content").with_boost(1.0))
        .add_field(FieldSchema::keyword("tags").with_boost(3.0))
        .default_field("content")
        .build();
    
    // Create an index with the schema
    let mut index = IndexBuilder::new()
        .with_in_memory(true)
        .with_schema(schema)
        .build()?;
    
    // Add documents with fields
    let mut doc1_fields = HashMap::new();
    doc1_fields.insert("title".to_string(), "Introduction to Python".to_string());
    doc1_fields.insert("content".to_string(), "Python is a high-level programming language with simple syntax.".to_string());
    doc1_fields.insert("tags".to_string(), "programming python tutorial".to_string());
    index.add_document_with_fields(1, doc1_fields)?;
    
    let mut doc2_fields = HashMap::new();
    doc2_fields.insert("title".to_string(), "Rust Systems Programming".to_string());
    doc2_fields.insert("content".to_string(), "Rust provides memory safety without garbage collection.".to_string());
    doc2_fields.insert("tags".to_string(), "programming rust systems".to_string());
    index.add_document_with_fields(2, doc2_fields)?;
    
    // Multi-field search using QueryNode API
    let query = QueryNode::or(vec![
        QueryNode::term_with_boost("title", "programming", 5.0),
        QueryNode::term_with_boost("content", "programming", 1.0),
        QueryNode::term_with_boost("tags", "programming", 3.0)
    ]);
    
    // Create execution parameters
    let mut params = QueryExecutionParams {
        scoring_algorithm: ScoringAlgorithm::BM25,
        limit: Some(10),
        ..Default::default()
    };
    
    // Add field boosts
    let mut field_boosts = HashMap::new();
    field_boosts.insert("title".to_string(), 1.5);
    field_boosts.insert("tags".to_string(), 1.2);
    params.field_boosts = field_boosts;
    
    // Execute the query
    let results = index.execute_query(&query, Some(params))?;
    
    println!("Found {} documents matching the query", results.scored_docs.len());
    for (doc_id, score) in &results.scored_docs {
        println!("Document {}: Score {:.4}", doc_id, score);
    }
    
    Ok(())
}

Creating an Index​

In-Memory Index​

For temporary search needs, create an in-memory index:

# Python
index = Index(in_memory=True)
// Rust
let index = IndexBuilder::new()
    .with_in_memory(true)
    .build()?;

Persistent Index​

For persistent storage, specify a storage path:

# Python
index = Index(storage_path="/path/to/index")
// Rust
let index = IndexBuilder::new()
    .with_storage_path("/path/to/index")
    .build()?;

Adding Documents​

Documents are the core units of searchable content. Each document has a unique ID and content text.

Basic Document​

# Python
index.add_document(1, "Document content text")
// Rust
index.add_document(1, "Document content text")?;

With Metadata​

Add rich metadata to improve search relevance and result presentation:

# Python
index.add_document(
    doc_id=1,
    content="Document content text",
    metadata={
        "title": "Document Title",
        "author": "Author Name",
        "tags": ["tag1", "tag2"],
        "category": "Example",
        "language": "en"
    }
)
// Rust
use fast_inverted_index::common::DocumentMetadata;

index.add_document_with_metadata(
    1,
    "Document content text",
    DocumentMetadata {
        id: 1,
        title: "Document Title".to_string(),
        author: Some("Author Name".to_string()),
        tags: vec!["tag1".to_string(), "tag2".to_string()],
        category: Some("Example".to_string()),
        language: "en".to_string(),
        ..Default::default()
    }
)?;

Searching​

The library provides multiple ways to search for documents.

Simple Queries​

# Python - Single term query
results = index.term_query("search")

# AND query - documents containing all terms
results = index.and_query(["search", "engine"])

# OR query - documents containing any term
results = index.or_query(["search", "index"])
// Rust - Single term query
let query = Query::term("search");
let results = index.query(&query)?;

// AND query
let query = Query::and(vec!["search".to_string(), "engine".to_string()]);
let results = index.query(&query)?;

// OR query
let query = Query::or(vec!["search".to_string(), "index".to_string()]);
let results = index.query(&query)?;

Parsed Queries​

Parse query strings:

# Python
results = index.parse_query("search engine")  # AND query by default
// Rust
let query = Query::parse("search engine");  // AND query by default
let results = index.query(&query)?;

Ranked Search​

Get results with relevance ranking:

# Python
results = index.search("search engine", ranking_method="bm25")

# With field boosting
results = index.search_with_metadata(
    "search engine",
    ranking_method="bm25",
    boost_fields={"title": 2.0, "tags": 1.5},
    limit=10
)
// Rust
let query = Query::parse("search engine");

// BM25 ranking
let results = index.search_bm25(&query, None)?;

// With field boosting
let mut boosts = std::collections::HashMap::new();
boosts.insert("title".to_string(), 2.0);
boosts.insert("tags".to_string(), 1.5);
let results = index.search_bm25(&query, Some(boosts))?;

// TF-IDF ranking
let results = index.search_tfidf(&query, None)?;

Term Suggestions​

Generate autocomplete suggestions from term prefixes:

# Python
suggestions = index.suggest_terms("prog")

# Print suggestions
for suggestion in suggestions:
    print(f"{suggestion['term']} (score: {suggestion['score']})")
// Rust
let suggestions = index.suggest_terms("prog")?;

// Print suggestions
for suggestion in &suggestions {
    println!("{} (score: {})", suggestion.term, suggestion.score);
}

Index Management​

Statistics​

Get information about the index:

# Python
stats = index.stats()
print(f"Documents: {stats['document_count']}")
print(f"Terms: {stats['term_count']}")
print(f"Cache hit rate: {stats['cache_hit_rate']}")
// Rust
let stats = index.stats()?;
println!("Documents: {}", stats.document_count);
println!("Terms: {}", stats.term_count);
println!("Cache hit rate: {}", stats.cache_hit_rate);

Optimization​

Optimize the index for better performance:

# Python
index.optimize()
// Rust
index.optimize()?;

Backup​

Create a backup of the index:

# Python
index.backup("/path/to/backup")
// Rust
index.backup(std::path::PathBuf::from("/path/to/backup"))?;
